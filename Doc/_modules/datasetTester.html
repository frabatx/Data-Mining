"""Dataset generator
This module creates a dataset generated by trees.
The dataset is structured as follows:
    * Create a seed, a list of lists that contain n random integer.
    * Fills the seed with noise
    * Builds a tree from a sequence
    * Adds noise to the tree by adding random nodes
    * Generates the dataset

.. moduleauthor:: Nicol√≤ Merzi <nicolo.merzi@studenti.unitn.it>
.. moduleauthor:: Francesco Battista <francesco.battista@studenti.unitn.it>
"""

import random
import copy
import csv
from random import shuffle, randint
from Binary_Tree import BinaryTree



[docs]def createSequence(n_list):
    """Create a seed
    
    :param n_list: Number of lists inside the seed
    :type n_list: Integer
    :return: It rapresents the seed of a single tree
    :rtype: list

    Example: 
            * [[4],[7],[1],[5]]
            * [[1,8],[5],[3,9]]
            * [[4],[5,6],[8]]
    """

    aSeq = []
    for i in range(0, n_list):
        if random.random() < 0.5:
            aSeq.append([randint(0,9)])
        else:
            aSeq.append([randint(0,9), randint(0,9)])
    return aSeq




[docs]def seqFiller(seq, n_max):
    """The function fills a list with random numbers up to n_max elements.
    
    :param seq: lsit of lists
    :type seq: list

    :param n_max: number of elements inside list
    :type n_max: Integer

    :return: A copy of the original list, filled by random numbers
    :rtype: list
    """

    seqfilled = copy.deepcopy(seq) #deep copy original list and modify only the copy
    for i in seqfilled:
        while len(i) < n_max:
            i.append(randint(0,9))
    return seqfilled #return copy list - original list "seq" untouched




[docs]def seqTreeBuilder(aSeq):
    """Given a sequence, the function build a tree
    
    :param aSeq: Generic list
    :type aSeq: list
    :return: Binary Tree
    :rtype: Binary Tree
    """

    tree = BinaryTree(aSeq[0]) #root node
    for i in aSeq[1:]:
        tree.insertBottom(i)
    return tree 



[docs]def createList(max_seq_value, max_range):
    """Create a list of random values, this list is used as single random node in the tree.
    
    :param max_seq_value: Number of elements in each lists
    :type max_seq_value: integer

    :param max_range: Max value of the elements inside each lists
    :type max_range: integer

    :return: List of integers
    :rtype: list
    """

    list = []
    for i in range(0,max_seq_value):
        list.append(randint(1,max_range))
    return list



[docs]def addNodesTree(tree, tree_noise, max_seq_value):
    """The function inserts n random nodes in a tree, this nodes are used as noise inside the tree
    
    :param tree: The tree where the function has to append random nodes
    :type tree: Binary Tree

    :param tree_noise: Number of nodes to append in a tree
    :type tree_noise: Integer

    :param max_seq_value: Number of values in a list
    :type max_seq_value: Integer

    :return: New filled tree
    :rtype: Binary Tree
    """

    for i in range(0, tree_noise):

        aList = createList(max_seq_value, 9)
    
        if random.random() < 0.5 : 
            tree.insertTraversal(aList)
        else:
            parent = tree.insertHead(aList)
            tree = parent
    return tree

    

[docs]def randomTrees(max_seq_value, treeLenght):
    """Create a random tree
    
    :param max_seq_value: Number of values in a list
    :type max_seq_value: Integer

    :param treeLenght: Number of nodes
    :type treeLenght: Integer

    :return: The random tree
    :rtype: Binary Tree
    """
    
    aList = createList(max_seq_value, 9)
    tree = BinaryTree(aList)
    for i in range(1, treeLenght):
        aList = createList(max_seq_value, 9)
        tree.insertTraversal(aList)
    return tree



[docs]def records2(tree, t_id, r_id, aList):
    """Given a tree the function builds a dataset as follows: 
        * Record id
        * Tree id
        * Pair of nodes in a parent-child relationship
    
    :param tree: Tree
    :type tree: Binary Tree

    :param t_id: Non-negative integer, the tree id. Equal for all nodes belonging to the same tree
    :type t_id: Integer

    :param r_id: Non-negative integer, the record id is unique and consecutive
    :type r_id: Integer

    :param aList: The list that contains information about a tree
    :type aList: List

    :return: Recursive, it is used to print the entire dataset
    :rtype: Record
    """

    r_id = r_id + 1
    if tree.left != None:
        #print output to screen
        print(str(r_id) + ';' + str(t_id) + ';' , tree.data , tree.left.data)
        #save output in a list
        aList.append([r_id,t_id,tree.data, tree.left.data])
        r_id = records2(tree.left, t_id, r_id, aList)
    if tree.right != None:
        #print output to screen
        print(str(r_id) + ';' + str(t_id) + ';' , tree.data , tree.right.data)
        #save output in a list
        aList.append([r_id,t_id,tree.data, tree.right.data])
        r_id = records2(tree.right, t_id, r_id, aList)
    return r_id

[docs]def dataset_generator (minTree, maxTree, n_list, max_seq_value, n_sequences, tree_noise, random_trees):
    """Dataset generator. Once the parameters have been set, a dataset can be built.
        * A sequence is generated
        * The sequence is then filled with noise (other random numbers)
        * A tree is generated
        * Noise is added to the tree
        * A new record is built
        * Random trees are added (noise)
        * A new record is built
    
    :param minTree: Minimum number of trees generated per sequence
    :type minTree: Integer

    :param maxTree: Maximum number of trees generated per sequence. Has to be greater or equal than minTree
    :type maxTree: Integer

    :param n_list: Sequence lenght
    :type n_list: Integer

    :param max_seq_value: Number of elements inside a list
    :type max_seq_value: Integer

    :param n_sequences: Number of sequences generated
    :type n_sequences: Integer

    :param tree_noise: Number of nodes to add as noise in a tree
    :type tree_noise: Integer

    :param random_trees: Number of random trees between trees created with a sequence (noise)
    :type random_trees: Integer
    """

    r_id = 0
    t_id = 0
    treeList = []
    #Creating sequences
    for sequences in range(1, n_sequences + 1):
        seq = createSequence(n_list) #creating sequence
        print("\n","Sequence created: ", seq)
        #creating trees 
        n_tree = random.randint(minTree, maxTree)
        for n_trees in range(1, n_tree + 1):
            t_id = t_id + 1
            seqfilled = seqFiller(seq, max_seq_value) #filled sequence
            
            print("\n","Sequence filled: ", seqfilled)
            
            tree = seqTreeBuilder(seqfilled) #from filled sequence to tree
            noised_tree = addNodesTree(tree,tree_noise, max_seq_value) #adding noise
            
            print("\n","Tree: ", n_trees, "\n")
            print(noised_tree, "\n")
            
            r_id = records2(noised_tree, t_id, r_id, treeList)
            r_id = r_id - 1

            for i in range(0,random_trees):
                print("\n", "Random Tree")
                t_id = t_id + 1
                treeNoise = randomTrees(max_seq_value, n_sequences+tree_noise)
                r_id = records2(treeNoise, t_id, r_id, treeList)
                r_id = r_id - 1
            
    print_dataset(treeList)
